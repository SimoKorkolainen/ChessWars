26  public class AiPlayer implements Player {
27      private BattleMoveCalculator moveCalculator;
28      private int searchDepth;

30      private Set<ArmyColor> myTeam; //Koodia voi muokata siten, että myTeam on tarpeeton

32      private Battle copy;

34      private static final double LARGE = 9000000;

36      /**
37       * Konstruktori luo tekoälypelaajan.
38       * @param searchDepth haun syvyys
39       * @param color tekoälyn väri
40       * @param battle taistelu, jossa tekoäly taistelee
41       */
42      public AiPlayer(int searchDepth, ArmyColor color, Battle battle) {
43          this.searchDepth = searchDepth;
44          this.color = color;
45          this.myTeam = new HashSet<>();
46          this.myTeam.add(color);
47          this.battle = battle;
48          System.out.println("Ai " + searchDepth);  
49      }

51      /**
52       * Metodi alustaa tekoälyn joukkueen.
53       */
54      public void setUpTeam() {
55          for (Player i : battle.getPlayers()) {
56              if (battle.getTeam(color) == battle.getTeam(i.getColor())) {
57                  myTeam.add(i.getColor());
58              }
59          }
60      }
61      /**
62       * Metodi laskee tekoälyn seuraavan siirron
63       * Tekoäly pelaa aivan kuin kaikki eri tiimeissä olevat vastustajat
64       * olisivat liittoutuneet keskenään ja omien tiimiläisten siirrot olisivat
65       * tekoälyn itse päätettävissä.
66       * @return paras siirto
67       */
68      public Move alphaBeta() {
69          copy = BattleCopier.copy(battle);
70          
71          moveCalculator = new BattleMoveCalculator(copy);
72          
73          double value = -LARGE * 2;
74          Move best = null;
75          
76          List<EvalMove> evalMoves = EvalMove.calculateSortedEvalMoves(copy, myTeam);
77          Collections.reverse(evalMoves);
78          for (EvalMove i : evalMoves) {
79              System.out.println(i.getMove());
80              copy.doMove(i.getMove());
81              
82              if (BattleWinnerChecker.myTeamHasWon(battle, color)) {
83                  copy.undoLastMove();
84                  return i.getMove();
85              }
86  
87              double newValue = calculateValue(-LARGE - searchDepth, LARGE + searchDepth, searchDepth);
88              if (newValue > value) {
89                  value = newValue;
90                  best = i.getMove();
91              }
92              
93              copy.undoLastMove();
94          }
95          
96          return best;
97      }

99      private double maxValue(double alpha, double beta, int depth) {
100 
101         if (depth == 0) {
102             return AiEvaluator.evaluate(copy.getMap(), myTeam, true);
103         }
104         
105         double value = -LARGE * 2;
106         
107         List<EvalMove> evalMoves = EvalMove.calculateSortedEvalMoves(copy, myTeam);
108         Collections.reverse(evalMoves);
109         for (EvalMove i : evalMoves) {
110 
111             copy.doMove(i.getMove());
112             
113             double newValue;
114             
115             if (BattleWinnerChecker.myTeamHasWon(battle, color)) {
116                 newValue = LARGE + depth;
117             } else {
118                 newValue = calculateValue(alpha, beta, depth);
119             }
120 
121             value = Math.max(value, newValue);
122 
123             copy.undoLastMove();
124 
125             if (beta <= value) {
126                 return value;
127             }
128             
129             alpha = Math.max(alpha, value);
130         }
131 
132         return value;
133     }
134     
135     private double minValue(double alpha, double beta, int depth) {
136 
137         if (depth == 0) {
138             return AiEvaluator.evaluate(copy.getMap(), myTeam, true);
139         }
140         double value = LARGE * 2;
141         
142         List<EvalMove> evalMoves = EvalMove.calculateSortedEvalMoves(copy, myTeam);
143         
144         for (EvalMove i : evalMoves) {
145 
146             copy.doMove(i.getMove());
147             
148             double newValue;
149             
150             if (BattleWinnerChecker.myTeamHasLost(battle, color)) {
151                 newValue = -LARGE - depth;
152             } else {
153                 newValue = calculateValue(alpha, beta, depth);
154             }
155             
156             value = Math.min(value, newValue);
157             
158             copy.undoLastMove();
159             
160             if (alpha >= value) {
161                 return value;
162             }
163             
164             beta = Math.min(beta, value);
165 
166         }
167         return value;
168     }

170     @Override
171     public ArmyColor getColor() {
172         return color;
173     }
174 
175     private double calculateValue(double alpha, double beta, int depth) {
176         if (myTeam.contains(copy.nextColorToMove())) {
177             return maxValue(alpha, beta, depth - 1);
178         } 
179         return minValue(alpha, beta, depth - 1);  
180     }

182     public Set<ArmyColor> getMyTeam() {
183         return myTeam;
184     }
185     /**
186      * Metodi laskee tekoälyn seuraavan siirron ja toteuttaa sen.
187      * @param thread metodia kutsunut BattleMoveThread-olio
188      */
189     @Override
190     public void calculateAndExecuteMove(BattleMoveThread thread) { 
191         Move next = alphaBeta();
192         thread.executeMove(next);  
193     }
194     
195     @Override
196     public Player copy(Battle battle) {
197         return new AiPlayer(searchDepth, color, battle);
198     }
