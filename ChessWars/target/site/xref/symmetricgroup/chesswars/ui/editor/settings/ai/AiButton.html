6   package symmetricgroup.chesswars.ui.editor.settings;
7   
8   import java.awt.Color;
9   import java.awt.Insets;
10  import javax.swing.JButton;
11  import javax.swing.JToggleButton;
12  import symmetricgroup.chesswars.players.ArmyColor;
13  import symmetricgroup.chesswars.ui.editor.RoundRectTextButton;
14  
15  /**
16   * AiButton on nappi, jonka avulla voidaan valita editorissa ohjaako tekoäly väriä.
17   */
18  public class AiButton extends RoundRectTextButton {
19      private ArmyColor color;
20      private boolean aiIsOn;
21      public AiButton(ArmyColor color) {
22          super(30, 30, color.getDrawingColor(), new Color(30, 30, 30), "");
23          this.color = color;
24          this.aiIsOn = true;
25          super.setMargin(new Insets(1, 1, 1, 1));
26          updateText();
27      }
28  
29      public boolean isAiIsOn() {
30          return aiIsOn;
31      }
32  
33      public void setAiIsOn(boolean aiIsOn) {
34          this.aiIsOn = aiIsOn;
35      }
36      public void updateText() {
37          if (aiIsOn) {
38              super.setButtonText("on");
39              
40          } else {
41              super.setButtonText("off");
42          }
43      }
44  }
