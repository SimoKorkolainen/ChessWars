25  /**
26   * SaveButtonListener on SaveButton-napin kuuntelija.
27   */
28  public class SaveButtonListener implements ActionListener {
29      private UserControl control;
30      private Battle battle;
31      private EditorRoom editor;
32  
33      public SaveButtonListener(Battle battle, EditorRoom room) {
34          this.battle = battle;
35          this.editor = room;
36          this.control = new UserControl();
37      }
38  
39      
40      
41      
42      @Override
43      public void actionPerformed(ActionEvent e) {
44          Map<ArmyColor, AiButton> aiButtons = editor.getEditorPanel().getAiPanel().getAiButtons();
45          TeamSelectionPanel teamPanel = editor.getEditorPanel().getTeamPanel();
46          battle.setPlayers(new ArrayList<>());
47          for (ArmyColor i : aiButtons.keySet()) {
48              AiButton button = aiButtons.get(i);
49              
50              if (button.playerIsIn()) {
51                  ArmyColor color = button.getColor();
52                  int team = teamPanel.getButton(color).getTeam();
53                  System.out.println(color + " in team " + team);
54                  addPlayer(color, team, button.aiIsOn());
55              
56              }
57          }
58          
59          battle.getMap().setMapName(editor.getMapAndName().getMapName().getText());
60          System.out.println("Saved battle: " + battle.getMap().getMapName());
61          
62          SaveButton save = (SaveButton) e.getSource();
63          save.updateSaveButton();
64          BattleIO.saveBattle("battles/battleTest.txt", battle);
65      }
66      
67      private void addPlayer(ArmyColor color, int team, boolean ai) {
68          Player player;
69          if (ai) {
70              player = new AiPlayer(4, color, battle, true);
71          } else {
72              player = new UserPlayer(color, control);
73          }
74          battle.setTeam(color, team);
75          battle.addPlayer(player);
76      }
77      
78  }
