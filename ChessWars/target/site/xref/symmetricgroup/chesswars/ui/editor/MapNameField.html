11  import javax.swing.JLabel;

16  import symmetricgroup.chesswars.ui.editor.settings.MapNameFieldListener;

18  /**
19   * MapNameField on kartan nimen syöttämiseen käytetty tekstikenttä.
20   */
21  public class MapNameField extends JPanel {
22  
23      private String mapName; 
24      
25      public MapNameField(String mapName, MapNameFieldListener listener) {
26          this.mapName = mapName;
27          super.setLayout(new GridLayout(2, 1));
28          super.setOpaque(false);
29          super.setBorder(new EmptyBorder(5, 5, 5, 5));
30          createComponents(listener);
31      }
32      
33      public void createComponents(MapNameFieldListener listener) {
34          
35          super.add(new JLabel("Battle name"));
36          
37          JTextField field = new JTextField(mapName);
38          field.getDocument().addDocumentListener(listener);
39          
40          super.add(field);
41      }
42      
43  }
