8   import java.awt.Color;
9   import java.awt.Dimension;
10  import javax.swing.ImageIcon;
11  import javax.swing.JButton;
12  import javax.swing.JToggleButton;
13  import symmetricgroup.chesswars.terrain.Terrain;
14  import symmetricgroup.chesswars.ui.editor.RoundRectImageButton;
15  
16  /**
17   * TerrainSelectionButton on editorissa maaston tyypin valitsemiseen k√§yttetty nappi.
18   */
19  public class TerrainSelectionButton extends RoundRectImageButton {
20      private Terrain terrain;
21  
22      public TerrainSelectionButton(Terrain terrain) {
23          super(1, terrain.getImage());
24          this.terrain = terrain;
25          updateIcons();
26      }
27  
28      public Terrain getTerrain() {
29          return terrain;
30      }
31      @Override
32      public Dimension getPreferredSize() {
33          return new Dimension(50, 50);
34      }
35      
36      @Override
37      public Dimension getMaximumSize() {
38          return getPreferredSize();
39      }
40      
41      public void updateIcons() {
42          
43          super.setIcon(new ImageIcon(icon(lighten(Color.gray, 20), lighten(Color.gray, 40))));
44          super.setPressedIcon(new ImageIcon(icon(lighten(Color.gray, 50), lighten(Color.gray, 70))));
45          super.setSelectedIcon(new ImageIcon(icon(lighten(Color.gray, 50), lighten(Color.gray, 70))));
46      }
47      
48  }
