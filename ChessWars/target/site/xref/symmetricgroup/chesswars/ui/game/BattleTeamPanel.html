8   import java.awt.Color;
9   import java.awt.Dimension;
10  import java.awt.Graphics;
11  import java.awt.Graphics2D;
12  import java.util.List;
13  import javax.swing.JPanel;
14  import symmetricgroup.chesswars.battle.Battle;
15  import symmetricgroup.chesswars.players.ArmyColor;
16  import symmetricgroup.chesswars.players.Player;
17  import symmetricgroup.chesswars.players.ai.AiPlayer;
18  import symmetricgroup.chesswars.ui.TextDrawer;
19  import symmetricgroup.chesswars.ui.editor.RoundRectTextButton;
20  
21  /**
22   *
23   * @author Simo
24   */
25  public class BattleTeamPanel extends JPanel {
26      private int width;
27      private int height;
28      private Battle battle;
29  
30      public BattleTeamPanel(Battle battle) {
31          this.battle = battle;
32          this.width = 100;
33          this.height = 20;
34          super.setDoubleBuffered(true);
35      }
36  
37      @Override
38      public void paintComponent(Graphics g) {
39          Graphics2D g2d = (Graphics2D) g;
40          List<Player> players = battle.getPlayers();
41  
42          g2d.setColor(Color.LIGHT_GRAY);
43  
44  
45          Color fill = RoundRectTextButton.lighten(ArmyColor.BLACK.getDrawingColor(), 50);
46  
47          TextDrawer drawer = new TextDrawer(fill, height * 2 / 3);
48  
49          int x = width / 2;
50  
51          int y = height;
52  
53  
54  
55          drawer.draw("Teams:", x, y, g2d);
56  
57          for (int i = 0; i < players.size(); i++) {
58              Player player = players.get(i);
59  
60              String name = player.getColor().toString();
61  
62              name += " " + battle.getTeam(player.getColor());
63  
64  
65              x = width / 2;
66  
67              y = (i + 2) * height + height / 2;
68  
69              fill = RoundRectTextButton.lighten(player.getColor().getDrawingColor(), 50);
70              drawer = new TextDrawer(fill,  height * 2 / 3);
71  
72              drawer.draw(name, x, y, g2d);
73          }
74  
75      }
76      
77      @Override
78      public Dimension getPreferredSize() {
79          return new Dimension(width, height * (battle.getPlayers().size() + 3));
80      }
81      
82  }
