6   package symmetricgroup.chesswars.ui;

8   import java.awt.BasicStroke;
9   import java.awt.Color;
10  import java.awt.Dimension;
11  import java.awt.Font;
12  import java.awt.FontMetrics;
13  import java.awt.Graphics;
14  import java.awt.Graphics2D;
15  import java.awt.RenderingHints;
16  import javax.swing.JPanel;
17  import javax.swing.border.EmptyBorder;
18  
19  /**
20   *
21   * @author Simo
22   */
23  public class RoundRectPanel extends JPanel {
24      private Color outline;
25      private Color fill;
26      private int border;
27      private JPanel inner;
28  
29      public RoundRectPanel(Color outline, Color fill, int border, JPanel inner) {
30          this.outline = outline;
31          this.fill = fill;
32          this.border = border;
33          this.inner = inner;
34          super.add(inner);
35      }
36      
37      
38      public void paintComponent(Graphics g) {
39          
40          Graphics2D g2d = (Graphics2D) g;
41          
42          RenderingHints hints = new RenderingHints(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
43          g2d.addRenderingHints(hints);
44          
45          Dimension d = inner.getSize();
46          
47          int width = (int) Math.round(d.getWidth()) + border * 2;
48          int height = (int) Math.round(d.getHeight()) + border * 2;
49          
50          
51          int offset = 2;
52          g2d.setColor(fill);
53          g2d.fillRoundRect(offset + border, offset + border, width - 2 * offset - 2 * border, height - 2 * offset - 2 * border, height / 3, height / 3);
54  
55          g2d.setColor(outline);
56          g2d.setStroke(new BasicStroke(3f));
57          g2d.drawRoundRect(offset + border, offset + border, width - 2 * offset - 2 * border, height - 2 * offset - 2 * border, height / 3, height / 3);
58  
59      }
60  }
