6   package symmetricgroup.chesswars.battle;
7   
8   import java.util.ArrayList;
9   import java.util.List;
10  import symmetricgroup.chesswars.battle.Battle;
11  import symmetricgroup.chesswars.battle.move.Move;
12  import symmetricgroup.chesswars.map.MapCopier;
13  import symmetricgroup.chesswars.players.Player;
14  
15  /**
16   * BattleCopier on tekoälyn käyttöön tarkoitettu luokka, jonka avulla on mahdollista kopioida Battle-olioita lähes kokonaan.
17   * Kopiolla ja alkuperäisellä Battle-oliolla voi olla samoja olioita attribuutteina.
18   */
19  public class BattleCopier {
20      
21      /**
22       * Metodi kopio osittain Battle-olion. Metodi copy ei kopioi Battle-oliota täydellisesti,
23       * sillä kopiolla ja alkuperäisellä on yhteisiä olioita.
24       * 
25       * @param battle kopiotava taistelu
26       * @return kopioitu taistelu
27       */
28      public static Battle copy(Battle battle) {
29          
30          Battle copy = new Battle(MapCopier.copy(battle.getMap()));
31          copy.setTeam(battle.getTeam());
32          copy.setTurn(battle.getTurn());
33          
34          copy.setMoves(copyMoves(battle.getMoves()));
35          
36          copy.setPlayers(copyPlayers(battle.getPlayers(), copy));
37          
38          return copy;
39      }
40      
41      
42      private static List<Move> copyMoves(List<Move> moves) {
43          List<Move> copy = new ArrayList<>();
44          
45          for (Move i : moves) {
46          
47              copy.add(i.copy());
48          
49          }
50          return copy;
51      }
52      
53      
54      private static List<Player> copyPlayers(List<Player> players, Battle reference) {
55          List<Player> copy = new ArrayList<>();
56          
57          for (Player i : players) {
58          
59              copy.add(i.copy(reference));
60          
61          }
62          return copy;
63      }
64  }
