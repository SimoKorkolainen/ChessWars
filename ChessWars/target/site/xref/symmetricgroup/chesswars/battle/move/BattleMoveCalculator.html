6   package symmetricgroup.chesswars.battle.move;
7   
8   import java.util.ArrayList;
9   import java.util.List;
10  import symmetricgroup.chesswars.battle.Battle;
11  import symmetricgroup.chesswars.map.BattleMap;
12  import symmetricgroup.chesswars.pieces.Piece;
13  import symmetricgroup.chesswars.players.ArmyColor;
14  
15  /**
16   *
17   * @author Simo
18   */
19  public class BattleMoveCalculator {
20      private Battle battle;
21      private BattleMap map;
22      
23      /**
24       * Konstruktori luo taistelun siirtojen laskemiseen tarkoitetun olion
25       * @param battle taistelu, josta halutaan laskea siirtoja
26       */
27      public BattleMoveCalculator(Battle battle) {
28          this.battle = battle;
29          this.map = battle.getMap();
30      }
31      
32     
33      /**
34       * Metodi palauttaa seuraavaksi vuorossa olevan pejaajan
35       * kaikki mahdolliset siirrot.
36       * @return kaikki mahdolliset siirrot
37       */
38      public List<Move> allPossibleNextMoves() {
39          List<Move> all = new ArrayList<>();
40          
41          for (int i = 0; i < map.getWidth(); i++) {
42              for (int j = 0; j < map.getHeight(); j++) {
43                  
44                  addMovesInPosition(all, i, j); 
45                  
46              }
47          
48          }
49      
50          return all;
51      }
52      
53      private void addMovesInPosition(List<Move> moves, int i, int j) {
54          
55          Piece piece = map.getPiece(i, j);
56  
57          ArmyColor next = battle.nextColorToMove();
58  
59          if (piece != null && piece.getColor() == next) {
60  
61              moves.addAll(piece.getMoves(i, j, battle));
62  
63          }
64          
65      }
66  
67  }
