7   package symmetricgroup.chesswars.battle.defeat;
8   
9   import java.util.ArrayList;
10  import java.util.List;
11  import symmetricgroup.chesswars.battle.Battle;
12  import symmetricgroup.chesswars.battle.move.Move;
13  import symmetricgroup.chesswars.map.BattleMap;
14  import symmetricgroup.chesswars.pieces.Piece;
15  import symmetricgroup.chesswars.players.ArmyColor;
16  import symmetricgroup.chesswars.players.Player;
17  
18  /**
19   * DefeatState on luokka, johon talletetaan pelaajan häviötila.
20   * Luokan metodien avulla on mahdollista toteuttaa ja kumota pelaajan häviö.
21   * Tekoälyn toiminnan kannalta on välttämätöntä, että pelaajan häviö voidaan kumota.
22   */
23  public class DefeatState {
24      private Player player;
25      private int playerPos;
26      private List<Move> removeMove;
27      
28     
29      
30      /**
31       * Konstruktori luo lopputila olion.
32       * @param player pelaaja, jonka lopputila halutaan luoda
33       * @param playerPos pelaajan, indeksi taistelun pelaajalistalla ennen pelaajan häviötä.
34       */
35      public DefeatState(Player player, int playerPos) {
36          this.player = player;
37          this.playerPos = playerPos;
38          this.removeMove = new ArrayList<Move>();
39      }
40      
41      
42      
43      /**
44       * Metodi poistaa hävinneen pelaajan nappulat kartalta ja tallentaa poistossa käytetyt siirrot listaan,
45       * jotta ne voitaisiin kumota tarvittaessa.
46       * @param map kartta, josta nappulat halutaan poistaa. 
47       */
48      public void removeDefeated(BattleMap map) {
49          for (int i = 0; i < map.getWidth(); i++) {
50              for (int j = 0; j < map.getWidth(); j++) {
51  
52                  Piece piece = map.getPiece(i, j);
53                  if (piece == null) {
54                      continue;
55                  }
56                  
57                  if (piece.getColor() == player.getColor()) {
58                      
59                      Move move = new Move(i, j, i, j, null, piece);
60                      map.doMove(move);
61                      removeMove.add(move);
62                  
63                  }
64  
65              }
66          }
67      
68      }
69      /**
70       * Metodi kumoaa häviöön liittyvän nappuloiden poisto-operaation.
71       * 
72       * @param map kartta, josta poisto-operaatio halutaan kumota
73       */
74      public void putDefeatedBack(BattleMap map) {
75          for (Move i : removeMove) {
76                    
77              map.undoMove(i);
78          
79          }
80      
81      }
82      
83      
84  
85      public Player getPlayer() {
86          return player;
87      }
88  
89  
90      public int getPlayerPos() {
91          return playerPos;
92      }
93  
94      public List<Move> getDefeatedRemoveMoves() {
95          return removeMove;
96      }
97  
98      public void setRemoveMoves(List<Move> removeMove) {
99          this.removeMove = removeMove;
100     }
101     
102 
103     
104 }
