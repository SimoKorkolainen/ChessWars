6   package symmetricgroup.chesswars.util;
7   
8   import java.awt.Color;
9   import java.awt.image.BufferedImage;
10  
11  /**
12   *
13   * @author Simo
14   */
15  public class ImageColorer {
16      
17      public static BufferedImage color(BufferedImage mask, Color color) {
18          
19          BufferedImage colored = new BufferedImage(mask.getWidth(), mask.getHeight(), BufferedImage.TYPE_4BYTE_ABGR_PRE);
20          
21          for (int i = 0; i < mask.getWidth(); i++) {
22              for (int j = 0; j < mask.getHeight(); j++) {
23                  
24                  int oldCol = mask.getRGB(i, j);
25                  
26                  colored.setRGB(i, j, getColoredColor(oldCol, color));
27              }
28              
29          }
30          
31      
32          return colored;
33      }
34      private static int getAlpha(int col) {
35          return (col & 0xff000000) >> 24;
36      }
37      private static int getRed(int col) {
38          return (col & 0x00ff0000) >> 16;
39      }
40      private static int getGreen(int col) {
41          return (col & 0x0000ff00) >> 8;
42      }
43      private static int getBlue(int col) {
44          return col & 0x000000ff;
45      }
46      
47      private static int getColoredRed(int col, Color color) {
48          return (int) Math.floor(getRed(col) / 255.0 * color.getRed());
49      }
50      
51      private static int getColoredGreen(int col, Color color) {
52          return (int) Math.floor(getGreen(col) / 255.0 * color.getGreen());
53      }
54      
55      private static int getColoredBlue(int col, Color color) {
56          return (int) Math.floor(getBlue(col) / 255.0 * color.getBlue());
57      }
58      
59      private static int getColoredColor(int oldCol, Color color) {
60          int newA = getAlpha(oldCol);
61          int newR = getColoredRed(oldCol, color);
62          int newG = getColoredGreen(oldCol, color);
63          int newB = getColoredBlue(oldCol, color);
64  
65          return (newA << 24) | (newR << 16) | (newG << 8) | newB;
66                  
67      }
68      
69  
70  }
