6   package symmetricgroup.chesswars.util;
7   
8   import java.awt.Color;
9   import java.awt.image.BufferedImage;
10  
11  /**
12   *
13   * @author Simo
14   */
15  public class ImageColorer {
16      
17      public static BufferedImage color(BufferedImage mask, Color color) {
18          BufferedImage colored = new BufferedImage(mask.getWidth(), mask.getHeight(), BufferedImage.TYPE_4BYTE_ABGR_PRE);
19          
20          for (int i = 0; i < mask.getWidth(); i++) {
21              for (int j = 0; j < mask.getHeight(); j++) {
22                  int col = mask.getRGB(i, j);
23                  
24                  int alpha = (col & 0xff000000) >> 24;
25                  int r = (col & 0x00ff0000) >> 16;
26                  int g = (col & 0x0000ff00) >> 8;
27                  int b = (col & 0x000000ff);
28                  
29                  int newR = (int) Math.floor((double) r / 255 * color.getRed());
30                  int newG = (int) Math.floor((double) g / 255 * color.getGreen());
31                  int newB = (int) Math.floor((double) b / 255 * color.getBlue());
32                  
33                  int newCol = (alpha << 24) | (newR << 16) | (newG << 8) | newB;
34                  
35                  colored.setRGB(i, j, newCol);
36              }
37              
38          }
39          
40      
41          return colored;
42      }
