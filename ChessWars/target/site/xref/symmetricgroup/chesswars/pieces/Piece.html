60      public List<Move> getMoves(int x, int y, Battle battle) {
61          List<Move> moves = new ArrayList<>();
62          BattleMap map = battle.getMap();
63          
64          for (int i = 0; i < moveDirX.length; i++) {
65              int stepsLeft = moveLength;
66              for (int j = 1; j <= moveLength; j++) {
67                  
68                  int newX = x + j * moveDirX[i];
69                  int newY = y + j * moveDirY[i];
70                  
71                  if (!map.isInside(newX, newY)) {
72                      break;
73                  
74                  }
75                  
76                  Terrain terrain = map.getTerrain(newX, newY);
77                  
78                  if (terrain.getClass() == Mountains.class) {
79                      if (!getName().equals("Pawn")) {
80                          break;
81                      }
82                  }
83                  
84                  stepsLeft -= terrain.moveCost();
85                  
86                  if (stepsLeft < 0 && j > 1) {
87                      break;
88                  }
89                  
90                  Piece eaten = map.getPiece(newX, newY);
91                  if (eaten != null) {
92                      if (eatDir[i] && battle.getTeam(eaten.getColor()) != battle.getTeam(color)) {
93                          
94                          boolean eat = !lastEatenWasOwnPawn(battle) || lastEaterWasPawn(battle); 
95                          
96                          if (eat) {
97                              moves.add(new Move(x, y, newX, newY, this, eaten));
98                          }
99                          
100                     }
101                     
102                     break;
103                 }
104                 if (!mustEatDir[i]) {
105                     Move move = new Move(x, y, newX, newY, this, null);
106                     moves.add(move);
107                     
108 
109                 }
110                 
111             }
112         
113         }
114         
115         return moves;
116     }

123     private boolean isPawn() {
124         return "Pawn".equals(getName());
125     }
126 
127     private boolean lastEaterWasPawn(Battle battle) {
128         if (battle.getMoves().isEmpty()) {
129             return false;
130         }
131         
132         Move last = battle.getLastMove();
133         Piece eater = last.getPiece();
134         return eater.isPawn();
135     }
136 
137     private boolean lastEatenWasOwnPawn(Battle battle) {
138         if (battle.getMoves().isEmpty()) {
139             return false;
140         }
141         
142         Move last = battle.getLastMove();
143         Piece eaten = last.getEaten();
144         if (eaten == null) {
145             return false;
146         }
147         
148         if (!eaten.isPawn()) {
149             return false;
150         }
151         
152         return eaten.getColor() == getColor();
153         
154     }
