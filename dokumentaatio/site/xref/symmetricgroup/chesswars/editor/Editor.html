6   package symmetricgroup.chesswars.editor;
7   
8   import java.awt.GridBagConstraints;
9   import java.awt.GridBagLayout;
10  import java.awt.Insets;
11  import javax.swing.JPanel;
12  import symmetricgroup.chesswars.map.Battle;
13  import symmetricgroup.chesswars.map.BattleMap;
14  import symmetricgroup.chesswars.ui.MapMouseListener;
15  import symmetricgroup.chesswars.ui.MapScreen;
16  
17  /**
18   *
19   * @author Simo
20   */
21  public class Editor extends JPanel {
22      
23      private BattleMap map;
24      private MapEditor editor;
25      private MapScreen screen;
26      public Editor(BattleMap map) {
27          this.map = map;
28          
29          createComponents();
30      }
31  
32      public void createComponents() {
33          super.setLayout(new GridBagLayout());
34          
35          GridBagConstraints constraints = new GridBagConstraints();
36          
37          screen = new MapScreen(new Battle(map), null);
38          
39          constraints.anchor = GridBagConstraints.WEST;
40          constraints.insets = new Insets(10, 10, 10, 10);
41          constraints.gridx = 0;
42          constraints.gridy = 0;
43          
44  
45          super.add(screen, constraints);
46          
47          constraints.anchor = GridBagConstraints.EAST;
48          constraints.gridx = 1;
49          constraints.gridy = 0;
50          EditorPanel panel = new EditorPanel();
51          super.add(panel, constraints);
52          
53          constraints.anchor = GridBagConstraints.EAST;
54          constraints.gridx = 1;
55          constraints.gridy = 1;
56          
57          ClearButtonListener clearListener = new ClearButtonListener(map, screen);
58          ClearButton clearButton = new ClearButton();
59          clearButton.addActionListener(clearListener);
60          
61          super.add(clearButton, constraints);
62          
63          editor = new MapEditor(panel, map);
64          MapMouseListener listener = new MapMouseListener(screen, null);
65          listener.setEditor(editor);
66          
67          screen.addMouseListener(listener);
68          
69      }
70  }
