6   package symmetricgroup.chesswars.ui.editor;
7   
8   import java.awt.event.ActionEvent;
9   import java.awt.event.ActionListener;
10  import symmetricgroup.chesswars.battle.Battle;
11  import symmetricgroup.chesswars.battle.BattleIO;
12  import symmetricgroup.chesswars.map.BattleMap;
13  import symmetricgroup.chesswars.players.ArmyColor;
14  import symmetricgroup.chesswars.players.ai.AiPlayer;
15  import symmetricgroup.chesswars.players.ui.UserControl;
16  import symmetricgroup.chesswars.players.ui.UserPlayer;
17  
18  /**
19   * SaveButtonListener on SaveButton-napin kuuntelija.
20   */
21  public class SaveButtonListener implements ActionListener {
22  
23      private Battle battle;
24  
25      public SaveButtonListener(BattleMap map) {
26          battle = new Battle(map);
27          battle.addPlayer(new AiPlayer(4, ArmyColor.WHITE, battle));
28          battle.addPlayer(new UserPlayer(ArmyColor.BLACK, new UserControl()));
29      }
30      
31      
32      
33      @Override
34      public void actionPerformed(ActionEvent e) {
35          BattleIO.saveBattle("battles/battleTest.txt", battle);
36      }
37      
38  }
