8   import java.awt.Dimension;
9   import javax.swing.ImageIcon;
10  import javax.swing.JButton;
11  import javax.swing.JToggleButton;
12  import symmetricgroup.chesswars.terrain.Terrain;
13  
14  /**
15   * TerrainSelectionButton on editorissa maaston tyypin valitsemiseen k√§yttetty nappi.

17  public class TerrainSelectionButton extends JToggleButton {
18      private Terrain terrain;
19  
20      public TerrainSelectionButton(Terrain terrain) {
21          super();
22          this.terrain = terrain;
23          super.setIcon(new ImageIcon(terrain.getImage()));
24      }
25  
26      public Terrain getTerrain() {
27          return terrain;
28      }
29      @Override
30      public Dimension getPreferredSize() {
31          return new Dimension(50, 50);
32      }
33      
34      @Override
35      public Dimension getMaximumSize() {
36          return getPreferredSize();
37      }

40  }
