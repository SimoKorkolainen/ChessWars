6   package symmetricgroup.chesswars.pieces;
7   
8   import java.awt.Color;
9   import java.awt.image.BufferedImage;
10  import symmetricgroup.chesswars.players.ArmyColor;
11  import symmetricgroup.chesswars.util.ImageColorer;
12  import symmetricgroup.chesswars.util.ImageLoader;
13  
14  /**
15   *
16   * @author Simo
17   */
18  public class PieceImages {
19      
20      
21      private BufferedImage whiteImage;
22      private BufferedImage blackImage;
23      private BufferedImage redImage;
24      private BufferedImage greenImage;
25      private BufferedImage blueImage;
26      private BufferedImage yellowImage;
27      
28      public PieceImages(String pieceName) {
29          
30          String startPart = "images/";
31          String endPart = pieceName + ".png";
32          
33          whiteImage = ImageLoader.loadImage(startPart + "White" + endPart);
34          blackImage = ImageLoader.loadImage(startPart + "Black" + endPart);
35          
36          redImage = ImageColorer.color(whiteImage, new Color(255, 100, 100));
37          greenImage = ImageColorer.color(whiteImage, new Color(100, 255, 100));
38          blueImage = ImageColorer.color(whiteImage, new Color(100, 100, 255));
39          yellowImage = ImageColorer.color(whiteImage, Color.yellow);
40          
41      }
42      
43      
44      public BufferedImage getImage(ArmyColor color) {
45  
46          switch (color) {
47              
48              case RED: return redImage;
49                  
50              case GREEN: return greenImage;
51                  
52              case BLUE: return blueImage;
53                  
54              case BLACK: return blackImage;
55                  
56              case WHITE: return whiteImage;
57              
58              case YELLOW: return yellowImage;
59                  
60              default: return null;
61      
62          }
63          
64      }
65      
66      
67  }
