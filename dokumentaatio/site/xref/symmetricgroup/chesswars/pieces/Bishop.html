6   package symmetricgroup.chesswars.pieces;
7   
8   import java.awt.image.BufferedImage;
9   import symmetricgroup.chesswars.players.ArmyColor;
10  
11  /**
12   * Bishop-luokka kuvaa lähettiä.
13   */
14  public class Bishop extends Piece {
15  
16      private static PieceImages images;
17  
18      public static final int[] MOVE_DIR_X = new int[]{1, -1, -1, 1};
19      public static final int[] MOVE_DIR_Y = new int[]{-1, -1, 1, 1};
20      public static final boolean[] EAT_DIR = new boolean[]{true, true, true, true};
21      public static final boolean[] MUST_EAT_DIR = new boolean[]{false, false, false, false};
22  
23      public Bishop(ArmyColor color) {
24          super(color, "Bishop");
25  
26          setMoveLength(6);
27  
28          setMoveDirX(MOVE_DIR_X);
29          setMoveDirY(MOVE_DIR_Y);
30          setEatDir(EAT_DIR);
31          setMustEatDir(MUST_EAT_DIR);
32  
33          if (images == null) {
34              images = new PieceImages(getName());
35          }
36      }
37  
38      @Override
39      public BufferedImage getImage() {
40          return images.getImage(getColor());
41      }
42      
43      @Override
44      public Piece copy() {
45          return new Bishop(getColor());
46      }
47  
48  }
